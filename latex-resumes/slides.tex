}

\frame{
	\frametitle{No, really. Why?}
	\begin{itemize}
		\item Because you {\bf don't} care about formatting
		\pause
		\item Because different employers want different things
		\pause
		\item Because you don't want your document to be editable
		\pause
		\item Because you {\bf do} write up professional reports and like them to look professional
		\pause
		\item Because you want your document to be readable on ANY machine
		\pause
		\item Talk nerdy to me
	\end{itemize}
}
